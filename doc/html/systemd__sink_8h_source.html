50  void sink_it_(const details::log_msg &msg) override
51  {
52  int err;
53  string_view_t payload;
54  memory_buf_t formatted;
55  if (enable_formatting_)
56  {
57  base_sink<Mutex>::formatter_->format(msg, formatted);
58  payload = string_view_t(formatted.data(), formatted.size());
59  }
60  else
61  {
62  payload = msg.payload;
63  }
64 
65  size_t length = payload.size();
66  // limit to max int
67  if (length > static_cast<size_t>(std::numeric_limits<int>::max()))
68  {
69  length = static_cast<size_t>(std::numeric_limits<int>::max());
70  }
71 
72  const string_view_t syslog_identifier = ident_.empty() ? msg.logger_name : ident_;
73 
74  // Do not send source location if not available
75  if (msg.source.empty())
76  {
77  // Note: function call inside '()' to avoid macro expansion
78  err = (sd_journal_send)("MESSAGE=%.*s", static_cast<int>(length), payload.data(), "PRIORITY=%d", syslog_level(msg.level),
79 #ifndef SPDLOG_NO_THREAD_ID
80  "TID=%zu", details::os::thread_id(),
81 #endif
82  "SYSLOG_IDENTIFIER=%.*s", static_cast<int>(syslog_identifier.size()), syslog_identifier.data(), nullptr);
83  }
84  else
85  {
86  err = (sd_journal_send)("MESSAGE=%.*s", static_cast<int>(length), payload.data(), "PRIORITY=%d", syslog_level(msg.level),
87 #ifndef SPDLOG_NO_THREAD_ID
88  "TID=%zu", details::os::thread_id(),
89 #endif
90  "SYSLOG_IDENTIFIER=%.*s", static_cast<int>(syslog_identifier.size()), syslog_identifier.data(), "CODE_FILE=%s",
91  msg.source.filename, "CODE_LINE=%d", msg.source.line, "CODE_FUNC=%s", msg.source.funcname, nullptr);
92  }
93 
94  if (err)
95  {
96  throw_spdlog_ex("Failed writing to systemd", errno);
97  }
98  }

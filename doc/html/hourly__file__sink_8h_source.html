24 /*
25  * Generator of Hourly log file names in format basename.YYYY-MM-DD-HH.ext
26  */
27 struct hourly_filename_calculator
28 {
29  // Create filename for the form basename.YYYY-MM-DD-H
30  static filename_t calc_filename(const filename_t &filename, const tm &now_tm)
31  {
32  filename_t basename, ext;
33  std::tie(basename, ext) = details::file_helper::split_by_extension(filename);
34  return fmt_lib::format(SPDLOG_FILENAME_T("{}_{:04d}-{:02d}-{:02d}_{:02d}{}"), basename, now_tm.tm_year + 1900, now_tm.tm_mon + 1,
35  now_tm.tm_mday, now_tm.tm_hour, ext);
36  }
37 };

39 /*
40  * Rotating file sink based on time.
41  * If truncate != false , the created file will be truncated.
42  * If max_files > 0, retain only the last max_files and delete previous.
43  */
44 template<typename Mutex, typename FileNameCalc = hourly_filename_calculator>
45 class hourly_file_sink final : public base_sink<Mutex>
46 {
47 public:
48  // create hourly file sink which rotates on given time
49  hourly_file_sink(
50  filename_t base_filename, bool truncate = false, uint16_t max_files = 0, const file_event_handlers &event_handlers = {})
51  : base_filename_(std::move(base_filename))
52  , file_helper_{event_handlers}
53  , truncate_(truncate)
54  , max_files_(max_files)
55  , filenames_q_()
56  {
57  auto now = log_clock::now();
58  auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(now));
59  file_helper_.open(filename, truncate_);
60  remove_init_file_ = file_helper_.size() == 0;
61  rotation_tp_ = next_rotation_tp_();
62 
63  if (max_files_ > 0)
64  {
65  init_filenames_q_();
66  }
67  }
68 
69  filename_t filename()
70  {
71  std::lock_guard<Mutex> lock(base_sink<Mutex>::mutex_);
72  return file_helper_.filename();
73  }
74 
75 protected:
76  void sink_it_(const details::log_msg &msg) override
77  {
78  auto time = msg.time;
79  bool should_rotate = time >= rotation_tp_;
80  if (should_rotate)
81  {
82  if (remove_init_file_)
83  {
84  file_helper_.close();
85  details::os::remove(file_helper_.filename());
86  }
87  auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(time));
88  file_helper_.open(filename, truncate_);
89  rotation_tp_ = next_rotation_tp_();
90  }
91  remove_init_file_ = false;
92  memory_buf_t formatted;
93  base_sink<Mutex>::formatter_->format(msg, formatted);
94  file_helper_.write(formatted);
95 
96  // Do the cleaning only at the end because it might throw on failure.
97  if (should_rotate && max_files_ > 0)
98  {
99  delete_old_();
100  }
101  }
102 
103  void flush_() override
104  {
105  file_helper_.flush();
106  }
107 
108 private:
109  void init_filenames_q_()
110  {
111  using details::os::path_exists;
112 
113  filenames_q_ = details::circular_q<filename_t>(static_cast<size_t>(max_files_));
114  std::vector<filename_t> filenames;
115  auto now = log_clock::now();
116  while (filenames.size() < max_files_)
117  {
118  auto filename = FileNameCalc::calc_filename(base_filename_, now_tm(now));
119  if (!path_exists(filename))
120  {
121  break;
122  }
123  filenames.emplace_back(filename);
124  now -= std::chrono::hours(1);
125  }
126  for (auto iter = filenames.rbegin(); iter != filenames.rend(); ++iter)
127  {
128  filenames_q_.push_back(std::move(*iter));
129  }
130  }
131 
132  tm now_tm(log_clock::time_point tp)
133  {
134  time_t tnow = log_clock::to_time_t(tp);
135  return spdlog::details::os::localtime(tnow);
136  }
137 
138  log_clock::time_point next_rotation_tp_()
139  {
140  auto now = log_clock::now();
141  tm date = now_tm(now);
142  date.tm_min = 0;
143  date.tm_sec = 0;
144  auto rotation_time = log_clock::from_time_t(std::mktime(&date));
145  if (rotation_time > now)
146  {
147  return rotation_time;
148  }
149  return {rotation_time + std::chrono::hours(1)};
150  }
151 
152  // Delete the file N rotations ago.
153  // Throw spdlog_ex on failure to delete the old file.
154  void delete_old_()
155  {
156  using details::os::filename_to_str;
157  using details::os::remove_if_exists;
158 
159  filename_t current_file = file_helper_.filename();
160  if (filenames_q_.full())
161  {
162  auto old_filename = std::move(filenames_q_.front());
163  filenames_q_.pop_front();
164  bool ok = remove_if_exists(old_filename) == 0;
165  if (!ok)
166  {
167  filenames_q_.push_back(std::move(current_file));
168  SPDLOG_THROW(spdlog_ex("Failed removing hourly file " + filename_to_str(old_filename), errno));
169  }
170  }
171  filenames_q_.push_back(std::move(current_file));
172  }
173 
174  filename_t base_filename_;
175  log_clock::time_point rotation_tp_;
176  details::file_helper file_helper_;
177  bool truncate_;
178  uint16_t max_files_;
179  details::circular_q<filename_t> filenames_q_;
180  bool remove_init_file_;
181 };

40 template<typename Mutex>
41 class kafka_sink : public base_sink<Mutex>
42 {
43 public:
44  kafka_sink(kafka_sink_config config)
45  : config_{std::move(config)}
46  {
47  try
48  {
49  std::string errstr;
50  conf_.reset(RdKafka::Conf::create(RdKafka::Conf::CONF_GLOBAL));
51  RdKafka::Conf::ConfResult confRes = conf_->set("bootstrap.servers", config_.server_addr, errstr);
52  if (confRes != RdKafka::Conf::CONF_OK)
53  {
54  throw_spdlog_ex(fmt_lib::format("conf set bootstrap.servers failed err:{}", errstr));
55  }
56 
57  tconf_.reset(RdKafka::Conf::create(RdKafka::Conf::CONF_TOPIC));
58  if (tconf_ == nullptr)
59  {
60  throw_spdlog_ex(fmt_lib::format("create topic config failed"));
61  }
62 
63  producer_.reset(RdKafka::Producer::create(conf_.get(), errstr));
64  if (producer_ == nullptr)
65  {
66  throw_spdlog_ex(fmt_lib::format("create producer failed err:{}", errstr));
67  }
68  topic_.reset(RdKafka::Topic::create(producer_.get(), config_.produce_topic, tconf_.get(), errstr));
69  if (topic_ == nullptr)
70  {
71  throw_spdlog_ex(fmt_lib::format("create topic failed err:{}", errstr));
72  }
73  }
74  catch (const std::exception &e)
75  {
76  throw_spdlog_ex(fmt_lib::format("error create kafka instance: {}", e.what()));
77  }
78  }
79 
80  ~kafka_sink()
81  {
82  producer_->flush(config_.flush_timeout_ms);
83  }
84 
85 protected:
86  void sink_it_(const details::log_msg &msg) override
87  {
88  producer_->produce(topic_.get(), 0, RdKafka::Producer::RK_MSG_COPY, (void *)msg.payload.data(), msg.payload.size(), NULL, NULL);
89  }
90 
91  void flush_() override
92  {
93  producer_->flush(config_.flush_timeout_ms);
94  }
95 
96 private:
97  kafka_sink_config config_;
98  std::unique_ptr<RdKafka::Producer> producer_ = nullptr;
99  std::unique_ptr<RdKafka::Conf> conf_ = nullptr;
100  std::unique_ptr<RdKafka::Conf> tconf_ = nullptr;
101  std::unique_ptr<RdKafka::Topic> topic_ = nullptr;
102 };
